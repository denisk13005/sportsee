/**
 *FETCH MAIN DATAS
 * @param {Number} userId id of user selected
 * @returns {object} userMainDatas
 */
export const fetchMainDatas = async (userId) => {
  const userMainDatas = await fetch(`http://localhost:5000/user/${userId}`)
    .then((res) => res.json())
    .then((data) => data)
  return userMainDatas
}
